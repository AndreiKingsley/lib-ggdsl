background

inline fun BarContextImmutable.background(crossinline block: BackgroundStyle.() -> Unit)

Sets background style for bars.

  • color - background color.

  • borderColor - background border color.

  • borderWidth - background border width. By default 0.

  • borderType - border type. By default solid.

  • borderRadius - background border radius. By default 0.

  • shadowBlur - background shadow blur.

  • shadowColor - background shadow color.

  • alpha - background opacity.

plot {
    bars {
        background {
            color(Color.GREY)
            borderColor(Color.BLACK)
            borderWidth(1.0)
            borderType(LineType.DASHED)
            borderRadius(1.3)
            shadowBlur(10.0)
            shadowColor(Color.GREEN)
            alpha(0.7)
        }
    }
}
